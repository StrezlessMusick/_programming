      --normal                  выводить нормальную разницу (по умолчанию)
  -q, --brief                   сообщать только о том, что файлы различаются
  -s, --report-identical-files  сообщать о том, что файлы одинаковые
  -c, -C NUM, --context[=NUM]   выводить NUM (по умолчанию 3) строк копий контекста
  -u, -U NUM, --unified[=NUM]   выводить NUM (по умолчанию 3) строк объединенного контекста
  -e, --ed                      выводить ed script
  -n, --rcs                     выводить RCS format diff
  -y, --side-by-side            выводить в две колонки
  -W, --width=NUM               выводить максимум NUM (по умолчанию 130) печатных колонок
      --left-column             выводить только левую колонку общих строк
      --suppress-common-lines   не выводить общие строки

  -p, --show-c-function         выводить в которой функции C присутствует каждое изменение
  -F, --show-function-line=RE   show the most recent line matching RE
      --label LABEL             использовать LABEL вместо имени файла
                                  (может повторяться)

  -t, --expand-tabs             раскрывать табуляцию в пробелы при выводе
  -T, --initial-tab             make tabs line up by prepending a tab
      --tabsize=NUM             размер табуляции NUM (по умолчанию 8) печатных колонок
      --suppress-blank-empty    подавлять пробелы и табуляцию в пустых строках при выводе (before empty output lines)
  -l, --paginate                пропустить вывод через 'pr' для разбивки его на страницы

  -r, --recursive                 рекурсивно сравнивать найденные поддиректории
      --no-dereference            не следовать по символьным ссылкам
  -N, --new-file                  обрабатывать отсутствующие файлы как пустые
      --unidirectional-new-file   обрабатывать отсутствующие первые файлы как пустые (treat absent first files as empty)
      --ignore-file-name-case     игнорировать регистр при сравнении имен файлов
      --no-ignore-file-name-case  проверять(consider) регистр при сравнении имен файлов
  -x, --exclude=PAT               исключить файлы, которые соответствуют PAT
  -X, --exclude-from=FILE         исключить файлы, которые соответствуют любому паттерну в FILE
  -S, --starting-file=FILE        начинать с FILE при сравнении директорий
      --from-file=FILE1           сравнивать FILE1 со всеми операндами;
                                    FILE1 может быть директорией
      --to-file=FILE2             сравнивать все операнды с FILE2;
                                    FILE2 может быть директорией

  -i, --ignore-case               игнорировать разницу в регистре в содержимом файлов
  -E, --ignore-tab-expansion      игнорировать различия в результате раскрытия табуляций
  -Z, --ignore-trailing-space     игнорировать пробелы в конце строки
  -b, --ignore-space-change       ignore changes in the amount of white space
  -w, --ignore-all-space          ignore all white space
  -B, --ignore-blank-lines        ignore changes where lines are all blank
  -I, --ignore-matching-lines=RE  ignore changes where all lines match RE

  -a, --text                      treat all files as text
      --strip-trailing-cr         strip trailing carriage return on input
      --binary                    read and write data in binary mode

  -D, --ifdef=NAME                output merged file with '#ifdef NAME' diffs
      --GTYPE-group-format=GFMT   format GTYPE input groups with GFMT
      --line-format=LFMT          format all input lines with LFMT
      --LTYPE-line-format=LFMT    format LTYPE input lines with LFMT
    These format options provide fine-grained control over the output
      of diff, generalizing -D/--ifdef.
    LTYPE is 'old', 'new', or 'unchanged'.  GTYPE is LTYPE or 'changed'.
    GFMT (only) may contain:
      %<  lines from FILE1
      %>  lines from FILE2
      %=  lines common to FILE1 and FILE2
      %[-][WIDTH][.[PREC]]{doxX}LETTER  printf-style spec for LETTER
        LETTERs are as follows for new group, lower case for old group:
          F  first line number
          L  last line number
          N  number of lines = L-F+1
          E  F-1
          M  L+1
      %(A=B?T:E)  if A equals B then T else E
    LFMT (only) may contain:
      %L  contents of line
      %l  contents of line, excluding any trailing newline
      %[-][WIDTH][.[PREC]]{doxX}n  printf-style spec for input line number
    Both GFMT and LFMT may contain:
      %%  %
      %c'C'  the single character C
      %c'\OOO'  the character with octal code OOO
      C    the character C (other characters represent themselves)

  -d, --minimal            try hard to find a smaller set of changes
      --horizon-lines=NUM  keep NUM lines of the common prefix and suffix
      --speed-large-files  assume large files and many scattered small changes

      --help               display this help and exit
  -v, --version            output version information and exit

FILES are 'FILE1 FILE2' or 'DIR1 DIR2' or 'DIR FILE...' or 'FILE... DIR'.
If --from-file or --to-file is given, there are no restrictions on FILE(s).
If a FILE is '-', read standard input.
Exit status is 0 if inputs are the same, 1 if different, 2 if trouble.
