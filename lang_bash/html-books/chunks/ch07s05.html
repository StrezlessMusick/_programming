Проверка знаний по теме "Проверка условий"

Общесистемный файл xinitrc может быть использован для запуска X-сервера. Данный файл содержит некоторое количество конструкций проверки условий if/then. Ниже следует выдержка из «древней» версии xinitrc (Red Hat 7.1 или где-то тех времён).

if [ -f $HOME/.Xclients ]; then
  exec $HOME/.Xclients
elif [ -f /etc/X11/xinit/Xclients ]; then
  exec /etc/X11/xinit/Xclients
else
     # установки по умолчанию. Не смотря на то, что мы не должны сюда попасть
     # (мы предоставляем возможность отката к Xclients), вреда от этого не будет.
     xclock -geometry 100x100-5+5 &
     xterm -geometry 80x50-50+150 &
     if [ -f /usr/bin/netscape -a -f /usr/share/doc/HTML/index.html ]; then
             netscape /usr/share/doc/HTML/index.html &
     fi
fi

Объясните работу конструкций проверки (test) в приведённом выше отрезке кода, затем изучите обновленную версию файла /etc/X11/xinit/xinitrc и проанализируйте конструкции if/then, присутствующие в нём. Возможно, Вам необходимо будет немного забежать вперёд к обсуждению команд grep, sed, и регулярных выражений.
